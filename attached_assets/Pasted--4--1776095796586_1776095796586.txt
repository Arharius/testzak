export function LegalPage() {
  const { type } = useParams()
  const content = LEGAL_CONTENT[type]

  if (!content) return (
    <div className="max-w-3xl mx-auto p-8 text-center text-gray-500">
      Страница не найдена
    </div>
  )

  return (
    <div className="max-w-3xl mx-auto p-8">
      <h1 className="text-2xl font-bold text-gray-900 mb-2">
        {content.title}
      </h1>
      <p className="text-sm text-gray-400 mb-8">
        Последнее обновление: {content.updated}
      </p>

      <div className="space-y-6">
        {content.sections.map((section, i) => (
          <div key={i}>
            <h2 className="font-semibold text-gray-800 mb-2">
              {section.heading}
            </h2>
            <div className="text-gray-600 text-sm leading-relaxed
                            whitespace-pre-line">
              {section.text}
            </div>
          </div>
        ))}
      </div>

      <div className="mt-12 pt-6 border-t border-gray-200
                      text-xs text-gray-400 text-center">
        По вопросам: info@[ваш-домен].ru
      </div>
    </div>
  )
}

  <Route path="/privacy" element={<LegalPage />} />

ШАГ 2.1 — Настроить отправку email (backend)

Добавить зависимость:
  pip install resend
  (или: pip install sendgrid / pip install python-dotenv smtplib)

Рекомендуется Resend.com — бесплатно до 3000 писем/мес,
простой API, хорошая доставляемость.

Получить API ключ на resend.com → добавить в .env:
  RESEND_API_KEY=re_xxxxxxxxx
  FROM_EMAIL=noreply@[ваш-домен].ru
  SITE_URL=https://[ваш-домен].ru

Создать backend/email_service.py:

import resend
import os
from datetime import datetime

resend.api_key = os.getenv("RESEND_API_KEY")
FROM_EMAIL = os.getenv("FROM_EMAIL", "noreply@tz-generator.ru")
SITE_URL = os.getenv("SITE_URL", "https://tz-generator.ru")


def send_welcome_email(to_email: str, user_name: str = None):
    """Отправить приветственное письмо при регистрации."""
    name = user_name or to_email.split("@")[0]
    try:
        resend.Emails.send({
            "from":    FROM_EMAIL,
            "to":      [to_email],
            "subject": "Добро пожаловать в ТЗ-генератор",
            "html":    build_welcome_html(name),
        })
        return True
    except Exception as e:
        print(f"[EMAIL ERROR] welcome to {to_email}: {e}")
        return False


def send_generation_complete_email(
    to_email: str,
    tz_title: str,
    download_url: str
):
    """Уведомление о готовности ТЗ."""
    try:
        resend.Emails.send({
            "from":    FROM_EMAIL,
            "to":      [to_email],
            "subject": f"ТЗ готово: {tz_title[:60]}",
            "html":    build_ready_html(tz_title, download_url),
        })
        return True
    except Exception as e:
        print(f"[EMAIL ERROR] generation to {to_email}: {e}")
        return False

from backend.email_service import send_welcome_email
import asyncio

# В конце функции регистрации, после db.commit():
asyncio.create_task(
    asyncio.to_thread(
        send_welcome_email,

  send_welcome_email(user.email, user.email.split("@")[0])
